# ENTRAR NO BASH DO CONTAINER
  - docker ps 
    (pegar ID do container)
  - docker exec -it 987c39189286 /bin/bash
    (exit para sair)


# Medir a performance do Site (Ferramenta ApacheBench ou AB)
  - apt-get install apache2-utils

  - ab -c 10 -n 1000 http://127.0.0.1:8000/static/favicon/favicon.ico

    -c = numero de conexoes
    -n = numero de requisicoes


# BACKUP / RESTORE POSTGRESQL

Backup:
docker exec -t -u postgres your-db-container pg_dumpall -c > dump_`date +%d-%m-%Y"_"%H_%M_%S`.sql

Restore:
cat your_dump.sql | docker exec -i your-db-container psql -Upostgres
